Use text blocks in a dashboard to include Markdown or unformatted text
as a block, useful for README content and other exposition.

\hypertarget{usage}{%
\subsection{Usage}\label{usage}}

\begin{Shaded}
\begin{Highlighting}[]
\FunctionTok{layout}\KeywordTok{:}
\AttributeTok{  }\FunctionTok{row}\KeywordTok{:}

\AttributeTok{    }\FunctionTok{title}\KeywordTok{:}\AttributeTok{ }\StringTok{\textquotesingle{}Introduction\textquotesingle{}}

\end{Highlighting}

\begin{itemize}
\tightlist
\item
  The file content will be parsed as Markdown and displayed in the box.
\item
  Row height will auto-fit to match the length of the text for long text
  blocks.
\end{itemize}
